// src/components/MlControl.tsx
import React from 'react';
import { motion } from 'framer-motion';
import { Minus, Plus } from 'lucide-react';

interface MlControlProps {
    ml: number;
    min: number;
    max: number;
    step?: number;
    onChange: (value: number) => void;
}

export const MlControl: React.FC<MlControlProps> = ({ ml, min, max, step = 50, onChange }) => (
    <div className="flex flex-col items-center w-1/3">
        <label className="mb-1 text-sm">Menge (ml)</label>
        <input
            type="range"
            min={min}
            max={max}
            step={step}
            value={ml}
            onChange={e => onChange(parseInt(e.target.value, 10))}
            className="w-full"
        />
        <div className="flex justify-center items-center space-x-2 mt-2">
            <motion.button
                type="button"
                onClick={() => onChange(Math.max(min, ml - step))}
                whileHover={{ scale: 1.2 }}
                animate={{ scale: 1 }}
                transition={{ type: 'spring', stiffness: 300 }}
                className="p-3 rounded-full focus:outline-none border-2 border-muted"
            >
                <Minus size={16} className="text-muted" />
            </motion.button>
            <span className="text-sm">{ml} ml</span>
            <motion.button
                type="button"
                onClick={() => onChange(Math.min(max, ml + step))}
                whileHover={{ scale: 1.2 }}
                animate={{ scale: 1 }}
                transition={{ type: 'spring', stiffness: 300 }}
                className="p-3 rounded-full focus:outline-none border-2 border-muted"
            >
                <Plus size={16} className="text-muted" />
            </motion.button>
        </div>
    </div>
);
